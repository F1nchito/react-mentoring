export const fetchFilmDetails = (id) => {
    return function(dispatch){
        return fetch(`https://api.themoviedb.org/3/movie/${id}?api_key=0e17c92161eda91ae716ecd38c3a0e3b&language=en-US`)
        .then(res => res.json())
        .then(function (data) {
            dispatch(receiveFilmDetails(data));
        })
    }
};
